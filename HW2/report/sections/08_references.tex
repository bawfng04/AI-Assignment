\section{Tài liệu tham khảo / References}

\begin{enumerate}
    \item \textbf{Shannon, C. E. (1950).} \textit{Programming a Computer for Playing Chess.} Philosophical Magazine, 41(314), 256–275. Bài báo cơ bản giới thiệu Minimax cho game-playing.

    \item \textbf{Knuth, D. E., \& Moore, R. W. (1975).} \textit{An Analysis of Alpha-Beta Pruning.} Artificial Intelligence, 6(4), 293–326. Phân tích toán học chi tiết về Alpha-Beta Pruning.

    \item \textbf{Coulom, R. (2006).} \textit{Efficient Selectivity and Backup Operators in Monte-Carlo Tree Search.} In Computers and Games (pp. 72–83). Springer. Bài báo giới thiệu MCTS cho Computer Go.

    \item \textbf{Schaeffer, J. (2009).} \textit{The Games Computers Play.} CRC Press. Có chứa lịch sử toàn bộ Computer Chess và các kỹ thuật classic như alpha-beta, iterative deepening, transposition tables.

    \item \textbf{Levy, D., \& Newborn, M. (1991).} \textit{How Computers Play Chess.} W. H. Freeman. Giáo trình kinh điển về computer chess algorithms.

    \item \textbf{Russell, S., \& Norvig, P. (2020).} \textit{Artificial Intelligence: A Modern Approach (4th ed.).} Pearson. Tài liệu tham khảo toàn diện về AI, bao gồm game-playing agents và search algorithms.

    \item \textbf{Campbell, M. S., Hoane Jr, A. J., \& Hsu, F. H. (2002).} \textit{Deep Blue.} Artificial Intelligence, 134(1-2), 57–83. Bài báo mô tả Deep Blue computer, một bước ngoặt trong computer chess.

    \item \textbf{Swiechowski, J., Godlewski, B., Sawicki, B., \& Mańdziuk, J. (2022).} \textit{Recent Advances in General Game Playing.} Applied Sciences, 12(5), 2532. Tổng quan gần đây về game-playing agents.

    \item \textbf{Gelly, S., Wang, Y., Munos, R., \& Teytaud, O. (2012).} \textit{Modification of UCT with Patterns in Monte-Carlo Go.} Technical Report, Multi-disciplinary Research and Collaboration Laboratory. Cải tiến MCTS cho Go và ứng dụng trong chess engines gần đây.

    \item \textbf{Python Software Foundation.} \textit{python-chess: a Python chess library.} \url{https://python-chess.readthedocs.io/}. Thư viện chính được sử dụng trong dự án.

    \item \textbf{tkinter Documentation.} \textit{Tkinter — Python interface to Tcl/Tk.} \url{https://docs.python.org/3/library/tkinter.html}. Framework GUI dùng cho giao diện.

    \item \textbf{matplotlib Documentation.} \textit{Matplotlib: Visualization with Python.} \url{https://matplotlib.org/}. Thư viện dùng để vẽ biểu đồ so sánh hiệu suất.

    \item \textbf{Stockfish Team.} \textit{Stockfish: Open Source Chess Engine.} \url{https://stockfishchess.org/}. Engine tham khảo cho hiểu biết về các kỹ thuật engine hiện đại.

    \item \textbf{Computer Chess Club Forum.} \url{http://www.talkchess.com/}. Diễn đàn hoạt động nhất về computer chess, nơi các nhà phát triển engine trao đổi ideas.

    \item \textbf{Polyglot Opening Book.} \url{https://www.chessprogramming.org/Polyglot}. Định dạng standard cho opening books trong chess engines.
\end{enumerate}
