\section{Tài liệu tham khảo}
\sloppy

\begin{enumerate}
    \item \textbf{Holland, J. H. (1975).} \textit{Adaptation in Natural and Artificial Systems.} University of Michigan Press. Công trình nền tảng giới thiệu về Thuật toán Di truyền.

    \item \textbf{Goldberg, D. E. (1989).} \textit{Genetic Algorithms in Search, Optimization, and Machine Learning.} Addison-Wesley. Tài liệu kinh điển về các ứng dụng của GA trong tối ưu hóa.

    \item \textbf{Colorni, A., Dorigo, M., \& Maniezzo, V. (1992).} \textit{A Genetic Algorithm to Solve the Timetable Problem.} Computational Optimization and Applications. Nghiên cứu tiên phong trong việc áp dụng GA giải quyết bài toán xếp lịch.

    \item \textbf{Burke, E. K., Elliman, D. G., \& Weare, R. F. (1996).} \textit{A Hybrid Genetic Algorithm for Highly Constrained Timetabling Problems.} Phương pháp GA lai (Hybrid GA) cho các bài toán xếp lịch có ràng buộc chặt.

    \item \textbf{Pillay, N., \& Banzhaf, W. (2008).} \textit{An Analysis of the Application of Genetic Algorithms to Examination Timetabling.} Phân tích chi tiết hiệu quả của GA trong bài toán xếp lịch thi cử.

    \item \textbf{Burke, E. K., \& Petrovic, S. (2002).} \textit{Recent Research Directions in Automated Timetabling.} European Journal of Operational Research. Tổng quan các hướng nghiên cứu hiện đại về xếp lịch tự động.

    \item \textbf{Rossi-Doria, O., et al. (2002).} \textit{A Comparison of the Performance of Different Metaheuristics on the Timetabling Problem.} So sánh hiệu quả của các phương pháp metaheuristic khác nhau.

    \item \textbf{Russell, S., \& Norvig, P. (2020).} \textit{Artificial Intelligence: A Modern Approach (4th ed.).} Pearson. Giáo trình chuẩn về Trí tuệ Nhân tạo.

    \item \textbf{Deb, K. (2001).} \textit{Multi-Objective Optimization Using Evolutionary Algorithms.} John Wiley \& Sons. Phương pháp tối ưu hóa đa mục tiêu.

    \item \textbf{Python Documentation.} \textit{Python 3 Standard Library.} \url{https://docs.python.org/3/}. Tài liệu tham khảo về ngôn ngữ lập trình Python.

    \item \textbf{Streamlit Documentation.} \textit{Streamlit -- The fastest way to build data apps.} \url{https://streamlit.io/}. Framework hỗ trợ xây dựng giao diện tương tác.

    \item \textbf{Matplotlib Documentation.} \textit{Matplotlib: Visualization with Python.} \url{https://matplotlib.org/}. Thư viện hỗ trợ trực quan hóa dữ liệu.
\end{enumerate}
